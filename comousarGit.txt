#como actualizar el repositorio

####hacer solo una vez: clonar el repositorio
#abrir git bash
#ir a carpeta donde quieres guardar repositorio (cd)
git clone https://github.com/igarcia17/causal_inference.git (nombre de como quieres llamar al repositorio en tu ordenador, opcional)

## para actualizarlo
#te mueves a la carpeta del repositorio (cd)
git init
git pull # para actualizar tu carpeta, descargar los archivos modificados de la uni
git status #(ver status de tu repositorio)
git add . #para poner en la cola tu version de los archivos
git commit -m 'mensaje de tu actualizacion' #para declarar que cambios has hecho
git push #para subir tus cambios
